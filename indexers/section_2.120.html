2.120 Temperature Measurement (TSENS)

Using the Library

```C

include // Defines NULL

include // Defines true

include // Defines EXIT_FAILURE

include

include "definitions.h" // SYS function prototypes

define TEMP_RESOLUTION (100U)

uint32_t result; volatile bool result_ready; float temp;

void result_handler(TSENS_STATUS status, uintptr_t context ) { result = TSENS_ConversionResultGet(); result_ready = true; }

// *********** // *********** // Section: Main Entry Point // *********** // ***********

int main ( void ) { / Initialize all modules / SYS_Initialize ( NULL );

printf("\n\r---------------------------------------------------------");
printf("\n\r                    TSENS Temperature Measurement                 ");
printf("\n\r---------------------------------------------------------\n\r");

TSENS_Enable();
TC0_TimerStart();
TSENS_CallbackRegister(result_handler, (uintptr_t) NULL);

while (1)
{
    /* Wait till conversion result is available */
    if (result_ready)
    {
         /* Read the result */
        result_ready = false;
        temp = (float)result / TEMP_RESOLUTION;

        printf(" TSENS Count = 0x%x    Temperature = %d.%02d celcius \r", (int)result, (int)temp, (int)((temp - (int)temp)*100.0));
    }
}

/* Execution should not come here during normal operation */

return ( EXIT_FAILURE );

}
